NAME

     CLAHE - performs contrast limited adaptive histogram equalization on an image.
 

SYNOPSIS

      #include <stdio.h>
      #include "CVIPimage.h"
      #include "CVIPdef.h"
      #include "CVIPcolor.h"
      #include "CVIPhisto.h"
 
     Image *CLAHE(Image *inputImage, int blocksize, int clipPercent, int bandSelect)
 
      <inputImage> - pointer to an Image structure
      <blocksize> - desired blocksize
      <clipPercent> - desired clipping percent
      <bandSelect> - RGB band on which to calculate histogram (0,1,2)
 

PATH

     $CVIPtoolsHOME\CVIPC\Histogram\CLAHE.c
 

DESCRIPTION

     This function performs a contrast limited adaptive histogram equalization algorithm. 
The algorithm calculates the CDF for the neighborhood of each pixel based on the blocksize. 
Edges are kept using a DCT symmetry method. Images must be a byte value.
 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     A contrast limited adaptive histogram equalized image.
 

HISTORY

     History information recorded: None
 

EXAMPLE

      #include <CVIPtoolkit.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPconvert.h>
      #include "CVIPcolor.h"
      #include "CVIPhisto.h"
 
      void main()
      {
      Image *cvipImage;
      IMAGE_FORMAT format;
      char *inputfile, *outputfile;
      int rows, cols;
      setDisplay_Image("picture", "Default");
 
      print_CVIP("\n\t\tEnter the Input File Name:  ");
      inputfile =(char *) getString_CVIP();
      format = getFormat_CVIP(inputfile);
      cvipImage = read_Image(inputfile, 1);
      view_Image(cvipImage, inputfile);
 
      cvipImage =(Image *) CLAHE(cvipImage,16,0.05,0);
 
      print_CVIP("\n\t\tEnter the Output File Name:  ");
      outputfile = getString_CVIP();
      view_Image(cvipImage, outputfile);
      write_Image(cvipImage,outputfile,CVIP_NO,CVIP_NO,format,1);
      free(inputfile);
      free(outputfile);
      }
